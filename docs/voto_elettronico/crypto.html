 45    def _hash_pair(self, left_hash: bytes, right_hash: bytes) -> bytes:
 46        """Calcola l'hash della concatenazione di due hash figli.
 47
 48        Args:
 49            left_hash (bytes): L'hash del nodo figlio sinistro.
 50            right_hash (bytes): L'hash del nodo figlio destro.
 51
 52        Returns:
 53            bytes: Il digest SHA-256 della coppia.
 54        """
 55        digest = hashes.Hash(hashes.SHA256())
 56        digest.update(left_hash + right_hash)
 57        return digest.finalize()

 59    def add_leaf(self, transaction_data: bytes) -> None:
 60        """Aggiunge una nuova foglia all'albero e ricalcola la Root Hash.
 61
 62        Args:
 63            transaction_data (bytes): La concatenazione (utente_enc || voto_enc).
 64                Se l'albero è sigillato (urne chiuse), l'inserimento viene rifiutato.
 65        
 66        Raises:
 67            RuntimeError: Se si tenta di aggiungere una foglia a un albero sigillato.
 68        """
 69        if self._is_sealed:
 70            raise RuntimeError("Operazione non consentita: Urne chiuse, albero sigillato.")
 71
 72        digest = hashes.Hash(hashes.SHA256())
 73        digest.update(transaction_data)
 74        leaf_hash = digest.finalize()
 75        
 76        new_node = MerkleNode(hash_val=leaf_hash, is_leaf=True)
 77        self.leaves.append(new_node)
 78        self._build_tree()
 79
 80    def _build_tree(self) -> None:
 81        """Ricalcola l'intero albero bottom-up a partire dalle foglie attuali.
 82        
 83        Se il numero di nodi a un certo livello è dispari, l'ultimo nodo
 84        viene duplicato per completare la coppia (padding crittografico standard).
 85        """
 86        if not self.leaves:
 87            self.root = None
 88            return
 89
 90        current_level = self.leaves
 91        
 92        while len(current_level) > 1:
 93            next_level = []
 94            # Processa a coppie
 95            for i in range(0, len(current_level), 2):
 96                left_node = current_level[i]
 97                
 98                # Se è dispari, duplica l'ultimo nodo
 99                if i + 1 < len(current_level):
100                    right_node = current_level[i + 1]
101                else:
102                    right_node = left_node
103                    
104                parent_hash = self._hash_pair(left_node.hash_val, right_node.hash_val)
105                parent_node = MerkleNode(hash_val=parent_hash, left=left_node, right=right_node)
106                next_level.append(parent_node)
107                
108            current_level = next_level
109            
110        self.root = current_level[0]
111

118        return self.root.hash_val if self.root else None

124    def get_proof_of_membership(self, transaction_index: int) -> list[tuple[bytes, str]]:
125        """Genera il percorso crittografico (Proof) per una determinata transazione.
126
127        Questo metodo permette la Verificabilità Individuale. Viene restituita la
128        lista degli hash "fratelli" necessari per ricalcolare la Root Hash a
129        partire dalla foglia specificata.
130
131        Args:
132            transaction_index (int): L'indice della foglia (transazione) nell'elenco.
133
134        Returns:
135            list[tuple[bytes, str]]: Una lista di tuple. Ogni tupla contiene 
136                l'hash fratello e la direzione ('left' o 'right') da cui concatenarlo.
137                
138        Raises:
139            IndexError: Se l'indice fornito non è valido.
140        """
141        if transaction_index < 0 or transaction_index >= len(self.leaves):
142            raise IndexError("Indice transazione non valido.")
143        
144        if not self.root or len(self.leaves) == 1:
145            return []
146
147        proof = []
148        current_index = transaction_index
149        num_nodes_at_level = len(self.leaves)
150        
151        # Scendiamo nell'albero per identificare i "fratelli" iterativamente
152        # Nota: L'implementazione qui è semplificata per la simulazione e si basa
153        # sulla conoscenza dell'indice per ricavare il partner a ogni livello.
154        
155        level_nodes = [node.hash_val for node in self.leaves]
156        
157        while len(level_nodes) > 1:
158            next_level = []
159            
160            # Se l'indice è pari, il fratello è a destra (se esiste)
161            if current_index % 2 == 0:
162                sibling_index = current_index + 1
163                if sibling_index < len(level_nodes):
164                    proof.append((level_nodes[sibling_index], 'right'))
165                else:
166                    # Caso dispari: il fratello è se stesso duplicato
167                    proof.append((level_nodes[current_index], 'right'))
168            # Se l'indice è dispari, il fratello è a sinistra
169            else:
170                sibling_index = current_index - 1
171                proof.append((level_nodes[sibling_index], 'left'))
172            
173            # Calcolo del livello superiore per il prossimo ciclo
174            for i in range(0, len(level_nodes), 2):
175                left_h = level_nodes[i]
176                right_h = level_nodes[i+1] if i+1 < len(level_nodes) else left_h
177                
178                digest = hashes.Hash(hashes.SHA256())
179                digest.update(left_h + right_h)
180                next_level.append(digest.finalize())
181            
182            level_nodes = next_level
183            current_index = current_index // 2
184            
185        return proof

188    def verify_proof(leaf_hash: bytes, proof: list[tuple[bytes, str]], expected_root: bytes) -> bool:
189        """Verifica una Proof of Membership in modo indipendente.
190
191        Questa funzione è stateless (statica) e simula il calcolo eseguito dal
192        client dell'utente per validare che il suo voto sia nel registro.
193
194        Args:
195            leaf_hash (bytes): L'hash calcolato dal client della propria tupla.
196            proof (list[tuple[bytes, str]]): Il path crittografico fornito dal server.
197            expected_root (bytes): La Root Hash pubblica firmata dalla Commissione.
198
199        Returns:
200            bool: True se la ricostruzione combacia con la Root Hash attesa.
201        """
202        current_hash = leaf_hash
203        
204        for sibling_hash, position in proof:
205            digest = hashes.Hash(hashes.SHA256())
206            if position == 'right':
207                digest.update(current_hash + sibling_hash)
208            else:
209                digest.update(sibling_hash + current_hash)
210            current_hash = digest.finalize()
211            
212        return current_hash == expected_root

62    def recover_secret(self, frammenti: list[tuple[int, int]]) -> int:
63        segreto_ricostruito = 0
64
65        for j, (x_j, y_j) in enumerate(frammenti):
66            numeratore = 1
67            denominatore = 1
68
69            for m, (x_m, _) in enumerate(frammenti):
70                if j == m:
71                    continue
72                numeratore = (numeratore * (-x_m)) % self.PRIME
73                denominatore = (denominatore * (x_j - x_m)) % self.PRIME
74
75            lagrange_poly = (y_j * numeratore * self._mod_inverse(denominatore, self.PRIME)) % self.PRIME
76            segreto_ricostruito = (segreto_ricostruito + lagrange_poly) % self.PRIME
77
78        return (segreto_ricostruito + self.PRIME) % self.PRIME
