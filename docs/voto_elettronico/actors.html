 21
 22    Mantiene il disaccoppiamento con l'identità reale gestendo esclusivamente
 23    le chiavi pubbliche degli utenti (pseudonimi). Mappa rigorosamente ogni
 24    pseudonimo a un singolo token per inibire il double-spending alla radice.

 26    Attributes:

 28        pk_ca2 (rsa.RSAPublicKey): Chiave pubblica dell'Autorità.

 31    """

 33    def __init__(self):
 34        """Inizializza CA2 generando le chiavi RSA e i database in memoria."""

 39    def _serializza_pk(self, pk_utente: rsa.RSAPublicKey) -> bytes:
 40        """Serializza una chiave pubblica in bytes per poterla usare come chiave nei dizionari.
 41
 42        Args:
 43            pk_utente (rsa.RSAPublicKey): La chiave pubblica da serializzare.
 44
 45        Returns:
 46            bytes: Rappresentazione PEM della chiave pubblica.
 47        """
 48        return pk_utente.public_bytes(
 49            encoding=serialization.Encoding.PEM,
 50            format=serialization.PublicFormat.SubjectPublicKeyInfo
 51        )

 53    def registra_pseudonimo_e_genera_token(
 54        self,
 55        pk_utente: rsa.RSAPublicKey,
 56        certificato_ca1: bytes,
 57        hash_h1_atteso: bytes,
 58        pk_ca1: rsa.RSAPublicKey
 59    ) -> tuple[bytes, bytes]:
 60        """Esegue la Registration Ceremony per lo pseudonimo digitale.
 61
 62        Verifica che l'utente sia stato autorizzato da CA1 controllandone la firma.
 63        Se valido e non duplicato, registra lo pseudonimo e rilascia un Token
 64        monouso firmato.
 65
 66        Args:
 67            pk_utente (rsa.RSAPublicKey): Lo pseudonimo digitale dell'elettore.
 68            certificato_ca1 (bytes): La firma di CA1 sul digest dell'utente.
 69            hash_h1_atteso (bytes): L'hash H1 in chiaro per validare la firma di CA1.
 70            pk_ca1 (rsa.RSAPublicKey): La chiave pubblica di CA1 per la verifica.
 71
 72        Raises:
 73            ValueError: Se la firma del certificato non è valida.
 74            ValueError: Se lo pseudonimo risulta già registrato.
 75
 76        Returns:
 77            tuple[bytes, bytes]: Una tupla contenente il token in chiaro e la sua 
 78            versione firmata digitalmente da CA2 (Token_Signed).
 79        """
 80        # 1. Verifica dell'autenticità del Certificato di Identità
 81        is_valid = verifica_firma(pk_ca1, certificato_ca1, hash_h1_atteso)
 82        if not is_valid:
 83            raise ValueError("Validazione fallita: Certificato CA1 non autentico o manomesso.")
 84
 85        pk_bytes = self._serializza_pk(pk_utente)
 86
 87        # 2. Controllo Unicità (Prevenzione identità duplicate/multi-voto)
 88        if pk_bytes in self.database_pseudonimi:
 89            raise ValueError("Pseudonimo già registrato. Tentativo di registrazione multipla bloccato.")
 90
 91        # 3. Generazione e salvataggio Token Monouso (32 bytes = 256 bits)
 92        token = os.urandom(32)
 93        self.database_pseudonimi[pk_bytes] = token
 94
 95        # 4. Apposizione della firma di CA2 sul token per garantirne l'origine
 96        token_signed = firma_messaggio(self.sk_ca2, token)
 97
 98        return token, token_signed

100    def genera_challenge(self, pk_utente: rsa.RSAPublicKey) -> bytes:
101        """Genera una challenge (nonce) per l'Authentication Ceremony (FIDO2).
102
103        Args:
104            pk_utente (rsa.RSAPublicKey): Lo pseudonimo che richiede l'accesso.
105
106        Raises:
107            KeyError: Se lo pseudonimo non è presente nel database.
108
109        Returns:
110            bytes: Un nonce crittografico casuale di 16 bytes.
111        """
112        pk_bytes = self._serializza_pk(pk_utente)
113        
114        if pk_bytes not in self.database_pseudonimi:
115            raise KeyError("Accesso Negato: Pseudonimo non riconosciuto da CA2.")
116
117        # Genera un Nonce fresco per garantire la validità limitata alla sessione
118        nonce = os.urandom(16)
119        self.sfide_attive[pk_bytes] = nonce
120        
121        return nonce
122
123    def verifica_challenge(self, pk_utente: rsa.RSAPublicKey, firma_challenge: bytes) -> bool:
124        """Verifica la risposta alla challenge firmata con la SK dell'Utente.
125
126        L'esito positivo dimostra il possesso fisico della chiave (User Presence)
127        e sventa tentativi di Replay Attack.
128
129        Args:
130            pk_utente (rsa.RSAPublicKey): Lo pseudonimo registrato.
131            firma_challenge (bytes): Il nonce firmato ricevuto dal client.
132
133        Raises:
134            ValueError: Se non ci sono challenge attive per lo pseudonimo.
135
136        Returns:
137            bool: True se l'autenticazione ha successo, False altrimenti.
138        """
139        pk_bytes = self._serializza_pk(pk_utente)
140        nonce_atteso = self.sfide_attive.get(pk_bytes)
141
142        if not nonce_atteso:
143            raise ValueError("Authentication Fallita: Nessuna challenge attiva per la sessione.")
144
145        # Verifica crittografica: solo la SK_Utente associata a PK_Utente può generare questa firma
146        is_valid = verifica_firma(pk_utente, firma_challenge, nonce_atteso)
147        
148        if is_valid:
149            # Invalida il nonce immediatamente per evitare Replay Attacks
150            del self.sfide_attive[pk_bytes]
151            return True
152            
153        return False

 29
 30    def __init__(self):
 31        """Inizializza la Commissione generando le proprie chiavi istituzionali."""
 32        self.sk_comm, self.pk_comm = genera_coppia_chiavi(config.RSA_KEY_SIZE)
 33        self.frammento_chiave: tuple[int, int] | None = None
 34        
 35        # Variabili per simulare l'incapsulamento della chiave privata dell'elezione.
 36        self.__sk_elezione_nascosta: rsa.RSAPrivateKey | None = None
 37        self.__segreto_sblocco: int | None = None
 38
 39    def avvia_setup_elezione(self) -> tuple[rsa.RSAPublicKey, tuple[int, int], tuple[int, int]]:
 40        """Esegue il setup crittografico dell'elezione (Fase 0).
 41
 42        Genera la coppia di chiavi dell'elezione. La chiave pubblica viene
 43        esposta per essere distribuita agli elettori. La chiave privata viene
 44        vincolata a un segreto frammentato tramite Shamir (t=2, n=3).
 45
 46        Returns:
 47            tuple: Contiene:
 48                - pk_elezione (rsa.RSAPublicKey): Da distribuire agli elettori.
 49                - frammento_ca1 (tuple): Frammento destinato a CA1.
 50                - frammento_ca2 (tuple): Frammento destinato a CA2.
 51        """
 52        # Generazione chiave specifica per crittografare i voti
 53        sk_elezione, pk_elezione = genera_coppia_chiavi(config.RSA_KEY_SIZE)
 54        
 55        # Generiamo un segreto casuale a 120 bit per il Secret Sharing
 56        # (rientra nel campo di Galois PRIME definito in shamir_sharing.py)
 57        segreto = random.SystemRandom().getrandbits(120)
 58        
 59        # Custodiamo la chiave, sbloccabile solo con il segreto ricostruito
 60        self.__sk_elezione_nascosta = sk_elezione
 61        self.__segreto_sblocco = segreto
 62        
 63        # Applica Shamir's Secret Sharing (Soglia=2, Frammenti=3)
 64        frammenti = split_secret(
 65            segreto=segreto, 
 66            soglia=config.SHAMIR_THRESHOLD, 
 67            n_frammenti=config.SHAMIR_SHARES
 68        )
 69        
 70        # Distribuzione logica dei frammenti
 71        self.frammento_chiave = frammenti[0]  # La Commissione conserva il frammento 1
 72        frammento_ca1 = frammenti[1]          # Destinato a CA1
 73        frammento_ca2 = frammenti[2]          # Destinato a CA2
 74        
 75        return pk_elezione, frammento_ca1, frammento_ca2
 76
 77    def ricostruisci_chiave(self, frammento_esterno: tuple[int, int]) -> rsa.RSAPrivateKey:
 78        """Ricostruisce la chiave dell'elezione tramite consenso crittografico.
 79
 80        Args:
 81            frammento_esterno (tuple[int, int]): Il frammento fornito da CA1 o CA2.
 82
 83        Raises:
 84            PermissionError: Se l'interpolazione fallisce (frammento errato).
 85
 86        Returns:
 87            rsa.RSAPrivateKey: La chiave privata pronta per la decifratura.
 88        """
 89        if not self.frammento_chiave:
 90            raise RuntimeError("La Commissione non ha generato o possiede il frammento.")
 91
 92        frammenti_per_consenso = [self.frammento_chiave, frammento_esterno]
 93        
 94        # Esegue l'interpolazione di Lagrange per trovare il termine noto P(0)
 95        segreto_calcolato = recover_secret(frammenti_per_consenso)
 96        
 97        if segreto_calcolato != self.__segreto_sblocco:
 98            raise PermissionError("Consenso fallito: Frammento non valido o insufficiente.")
 99            
100        return self.__sk_elezione_nascosta

102    def esegui_scrutinio(self, merkle_tree: MerkleTree, sk_elezione: rsa.RSAPrivateKey) -> dict[str, int]:
103        """Esegue l'apertura delle urne, decifrando i voti in chiaro.
104
105        Estrae le foglie dal Merkle Tree e decifra esclusivamente il crittogramma 
106        del voto (voto_enc). Il crittogramma dell'utente (utente_enc) viene 
107        ignorato, preservando rigorosamente lo pseudoanonimato.
108
109        Args:
110            merkle_tree (MerkleTree): Il registro pubblico sigillato.
111            sk_elezione (rsa.RSAPrivateKey): La chiave dell'elezione ricostruita.
112
113        Returns:
114            dict[str, int]: Un dizionario con i risultati aggregati dello scrutinio.
115        """
116        risultati = Counter()
117
